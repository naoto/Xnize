==================================

    $ mv sample/data.yaml ./

run:
    $ ./xnize.rb あれなあれがあれであれ

add:
    $ ./xnize.rb add 正規表現 置換文字

delete:
    $ ./xnize.rb del 正規表現

prefix add:
    $ ./xnize.rb prefix add ほげ

prefix del:
    $ ./xnize.rb prefix del ほげ

suffix add:
    $ ./xnize.rb suffix add ほげ

suffix del:
    $ ./xnize.rb suffix del ほげ

list:
    $ ./xnize.rb list
